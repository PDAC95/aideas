    **saveProfileName(firstName: string, lastName: string):**

    - Validate with profileSchema
    - Compute fullName = [firstName, lastName].filter(Boolean).join(' ')
    - UPDATE profiles SET first_name, last_name, full_name WHERE id = user.id (anon client — profiles has authenticated UPDATE policy)
    - Return `{ success: true }` or `{ error: string }`

    **saveCompanyName(orgId: string, companyName: string):**
    - Auth check
    - Verify org membership (query organization_members for user.id + orgId, role must be owner or admin)
    - Use getAdminClient() to UPDATE organizations SET name = companyName WHERE id = orgId (service_role needed — no authenticated UPDATE policy on organizations)
    - Return `{ success: true }` or `{ error: string }`

    **saveAvatarUrl(avatarUrl: string | null):**
    - Auth check
    - UPDATE profiles SET avatar_url = avatarUrl WHERE id = user.id (anon client)
    - Return `{ success: true }` or `{ error: string }`

    **saveHourlyCost(orgId: string, hourlyCost: number):**
    - Auth check
    - Validate with hourlyCostSchema
    - Role check: query organization_members, must be owner or admin. Return `{ error: 'unauthorized' }` if not.
    - Fetch current settings: admin.from('organizations').select('settings').eq('id', orgId).single()
    - Merge: `{ ...(org?.settings ?? {}), hourly_cost: hourlyCost }`
    - Update via admin client: admin.from('organizations').update({ settings: newSettings }).eq('id', orgId)
    - Return `{ success: true }` or `{ error: string }`

    **switchLocale(locale: 'en' | 'es'):**
    - Validate locale is 'en' or 'es'
    - const cookieStore = await cookies()
    - cookieStore.set('NEXT_LOCALE', locale, { path: '/', httpOnly: false, maxAge: 60 * 60 * 24 * 365, sameSite: 'lax' })
    - No return needed (void)

    **changePassword(currentPassword: string, newPassword: string, confirmPassword: string):**
    - Auth check
    - Validate with changePasswordSchema
    - Verify current password: supabase.auth.signInWithPassword({ email: user.email!, password: currentPassword })
    - If sign-in fails: return `{ error: 'wrong_current_password' }`
    - Update password: supabase.auth.updateUser({ password: newPassword })
    - If update fails: return `{ error: 'update_failed' }`
    - Return `{ success: true }`

    export async function fetchSettingsData(userId: string, orgId: string) {
      const supabase = await createClient();
      const [profileResult, orgResult, memberResult] = await Promise.all([
        supabase.from('profiles').select('first_name, last_name, full_name, avatar_url, email').eq('id', userId).single(),
        supabase.from('organizations').select('name, settings').eq('id', orgId).single(),
        supabase.from('organization_members').select('role').eq('user_id', userId).eq('organization_id', orgId).single(),
      ]);
      return {
        profile: profileResult.data ? {
          firstName: profileResult.data.first_name,
          lastName: profileResult.data.last_name,
          fullName: profileResult.data.full_name,
          avatarUrl: profileResult.data.avatar_url,

        org: {
          orgId,
          orgName: orgResult.data?.name ?? null,
          hourlyCost: (orgResult.data?.settings as Record<string, unknown>)?.hourly_cost as number | null ?? null,
          role: (memberResult.data?.role ?? 'viewer') as SettingsOrgData['role'],
        } as SettingsOrgData,
      };
    }
