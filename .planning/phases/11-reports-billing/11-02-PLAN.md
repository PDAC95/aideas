**Reports page — page.tsx:**
Create an RSC page that reads `searchParams.period` and fetches data.

```typescript
import { getTranslations } from "next-intl/server";
import { createClient } from "@/lib/supabase/server";
import { getOrgId, fetchReportsData } from "@/lib/dashboard/queries";
import { ReportsPeriodSelector } from "@/components/dashboard/reports-period-selector";
import dynamic from "next/dynamic";
// Dynamic imports for chart (Recharts needs ssr:false)

export default async function ReportsPage({
  searchParams,
}: {
  searchParams: Promise<{ period?: string }>;
}) {
  const { period = "this_month" } = await searchParams;
  const validPeriods = ["this_month", "last_month", "last_3_months"];
  const safePeriod = validPeriods.includes(period) ? period : "this_month";

  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return null;

  const orgId = await getOrgId(user.id);
  if (!orgId) return null;

  const data = await fetchReportsData(orgId, safePeriod);
  const t = await getTranslations("dashboard.reports");

  // Build translations objects for child components
  const periodTranslations = {
    this_month: t("period.this_month"),
    last_month: t("period.last_month"),
    last_3_months: t("period.last_3_months"),
  };

  const kpiTranslations = {
    tasksCompleted: t("kpi.tasksCompleted"),
    hoursSaved: t("kpi.hoursSaved"),
    estimatedValue: t("kpi.estimatedValue"),
    vsLastPeriod: t("kpi.vsLastPeriod"),
    noHourlyCost: t("kpi.noHourlyCost"),
    settingsLink: t("kpi.settingsLink"),
  };

  const chartTranslations = {
    title: t("chart.title"),
    empty: t("chart.empty"),
  };

  const breakdownTranslations = {
    title: t("breakdown.title"),
    automationName: t("breakdown.automationName"),
    metricLabel: t("breakdown.metricLabel"),
    count: t("breakdown.count"),
    hoursSaved: t("breakdown.hoursSaved"),
    total: t("breakdown.total"),
    viewAll: t("breakdown.viewAll"),
    showLess: t("breakdown.showLess"),
  };

  const emptyTranslations = {
    title: t("empty.title"),
    message: t("empty.message"),
    cta: t("empty.cta"),
  };
```

If `data` is null (no automations), render empty state: centered card with icon (BarChart3 from lucide-react), title, message, and Link to `/dashboard/automations`.

If `data` exists, render:
1. Page title `<h1>` with `t("title")`
2. `<ReportsPeriodSelector>` with `currentPeriod={safePeriod}` and `translations={periodTranslations}`
3. `<ReportsKpiCards>` with `kpi={data.kpi}`, `hourlyCost={data.hourlyCost}`, `translations={kpiTranslations}`
4. Dynamic `<ReportsWeeklyChart>` with `data={data.weeklyChart}`, `translations={chartTranslations}` — imported via `next/dynamic({ ssr: false })` with loading skeleton
5. `<ReportsBreakdownTable>` with `rows={data.breakdown}`, `translations={breakdownTranslations}`

Layout: vertical stack with gap-6 or gap-8, max-w same as other dashboard pages.
